================================================================================
WRITING STANDARDS
================================================================================

PROFESSIONAL LANGUAGE:
  ✅ "police arrested" (not "cops caught")
  ✅ "acting on information received" (not "got a tip")
  ✅ "scheduled to be produced before" (not "will go to court")
  ✅ "investigations are being conducted" (not "looking into it")
  ✅ "in possession of" (not "had")
  ✅ "along with" (not "with")

PERSON DETAILS:
  ✅ Title + Full Name + Age + Complete Address
  ✅ "Rev. Embilipitiye Indrarathana thero, aged 68"
  ✅ "aged 56 of # 896, Mayuragama, Sewanagala"
  ❌ "monk, 68 years old, from Embilipitiya"

ITEM DETAILS:
  ✅ Specific quantities: "80g of gunpowder", "2 electric detonators"
  ✅ Detailed descriptions: "locally made muzzle loading firearm"
  ❌ Vague terms: "some gunpowder", "a weapon"

DATE FORMAT:
  ✅ "On the 17th March 2026"
  ✅ "on the 18th of March 2026"
  ❌ "On March 17, 2026"
  ❌ "On 17/03/2026"
